C] Notes about building [Node web server] using , [express.js] Framework :

  • This lesson is building the express node web server with organizing the script by using the [MVC] pattern
  • By using the {MVC} pattern design => there are essential differneces between the defined the {main Route + handling it's return } , and defined it's {Route CRUD operations} as following :

      res.send('welcome to the express web server !') ;
      } ) ;



      2] [Second method ] : [Express.js + using MVC] pattern method => according to next steps :

      • a- Inside {routes/root.js} file => main routes defintion
        // 1- Importing requirments [express , router , path ] :
        express = require ('express') ;
        router = express.Router() ;
        path = require('path') ;

        res.sendFile(path.join( __dirname , 'views' , 'index.html') ) ;
        res.send('welcome to the express web server !') ;
        } ) ;

        module.exports = router

        1- CRUD operations of the of first Route { '/'} :

        1. First CRUD method [get opt] : [get opertaion] of all current employees data object , as to next steps :
          // -- Define the method of the handling [get opertaion] of all employees data object inside the current response json [ using the parmeter of the (res) ]:
          const getAllEmployees = (req, res) => {
          // Assign the obained data of the employees inside the {res} parameter [in json form ] :
          res.json(data.employees);
          }



        2. Second CRUD method [post/add new elment] opt: [create new log/element inside the current data and post/add it inside the current data json object ] , as next steps :
          const createNewEmployee = (req, res) => {
          // 1 - define an object of a new employee object [ including the incomming each field of new element's request] :
          const newEmployee = {
          id: data.employees[data.employees.length - 1].id + 1 || 1 , => [getting the (id) of current element/employee - using the data array's length & consitnial assgin - - ]
          firstname: req.body.firstname , => [ getting [firstname] value as a property of the (req) ]
          lastname: req.body.lastname => [ getting [lastname] value as a property of the (req) ]
          }

          // 2- handle the error of the unexisted value of one of boht fields of [first name] or [last name] of the inserted new element/employee - using the conditional + {res.status()} method :
          if (!newEmployee.firstname || !newEmployee.lastname) {
          return res.status(400).json({ 'message': 'First and last names are required...' });
          }

          // 3- Adding the new element object {newEmployee} [after handling the upper error ] into the upper defined object variable of [data.employees] => by using the {setEmployees} setter function by using the destructuring method :
          data.setEmployees([...data.employees, newEmployee] );

          // 4- Assign the final {data} value ( after adding the new object - newEmployee - ] ) into the [res] parameter , with use a new record code (201) : :
          res.status(201).json(data.employees) ;
          }



        3. Third CRUD method [put/update opt]: [ update a certain existed element in the current data ] , as next steps :
          const updateEmployee = (req, res) => {
          // 1- finding the requested element from the data object according to its {id} , usign the {find()} method and inner condition of the existed id :
          const employee = data.employees.find(emp => emp.id === parseInt(req.body.id)) ;

          // 2- handling Error of the requested element [id] unexisted/not found , by Sending an error status code (400) :
          if (!employee) {
          return res.status(400).json({ "message": `Employee ID ${req.body.id} not found ! ` });
          }

          // 3- check if the [firstname requested element] is existed , and assign the new element's firstname {employee.firstname} value with the value of the obtained value from request {req.body.firstname} :
          if (req.body.firstname) employee.firstname = req.body.firstname ;
          // 4- check if the [lastname requested element] is existed , and assign the new element's lastname {employee.lastname} value with the value of the obtained value from request {req.body.lastname} :
          if (req.body.lastname) employee.lastname = req.body.lastname ;
          // 5- define a variable of filtered array of all existed elements - except the current updated helement - [ using the filter() ] :
          const filterArray = data.employees.filter(emp => emp.id !== parseInt(req.body.id));
          // 6- define an unsorted array as accumaltated the upper defined array of the all existed elements concateanated with the current updated element : :
          const unsortedArray = [ ...filterArray , employee ] ;
          // 7- adding the [upper accumaltated array] into the [current main data object] , by using the setter function + with assign the proper [id] value using the sort() method :
          data.setEmployees( unsortedArray.sort( (a, b) => a.id > b.id ? 1 : a.id < b.id ? -1 : 0 ) ) ;
          // 8- sending the [employees] property of [new data] after being filled -> into the response parameter in json form ::
          res.json( data.employees ) ;
          }



        4. Fourth CRUD method [ delete/remove opt]: [delete a certain element from the data json ] :
          const deleteEmployee = (req, res) => {
          // 1- finding the requested element from the data object according to its {id} , usign the {find()} method and inner condition of the existed id :
          const employee = data.employees.find(emp => emp.id === parseInt(req.body.id)) ;

          // 2- handling Error of the requested element [id] unexisted/not found , by Sending an error status code (400) :
          if (!employee) {
          return res.status(400).json({ "message": `Employee ID ${req.body.id} not found ! ` });
          }

          // 3- define an filtered array of all existed elements - except the current requested element - [ using the filter() ] :
          const filterArray = data.employees.filter(emp => emp.id !== parseInt(req.body.id));
          // 4- set/assign the upper defined [filtered array ] of all existed elements (only) inside the data [ using the setter function + destrucurtring method ] :
          data.setEmployee([...filterArray]);
          // 5- sending the [employees] property object (after being filled with filterd array ) into the response parameter (res) [ with json form] :
          res.json( data.employees );
          }

        3- sending the [requested obtained employee object only ] after being found into the response parameter as json format :
